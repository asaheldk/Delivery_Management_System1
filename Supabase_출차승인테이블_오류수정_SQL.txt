-- Supabase Dashboard → SQL Editor → New query에 이 파일 전체를 붙여 넣고
-- 오른쪽 아래 Run 버튼을 한 번만 누르세요.
-- 기존 배송원·거래처·발주·완료 자료는 삭제하거나 변경하지 않습니다.

begin;

create table if not exists public.departure_checks (
  driver_id uuid not null references public.profiles(id) on delete cascade,
  work_date date not null,
  vehicle_number integer not null check (vehicle_number between 1 and 999),
  orders_checked boolean not null default false,
  tanks_checked boolean not null default false,
  boxes_checked boolean not null default false,
  urgent_checked boolean not null default false,
  handover_checked boolean not null default false,
  driver_ready boolean not null default false,
  approved boolean not null default false,
  approved_by uuid references public.profiles(id) on delete set null,
  approved_at timestamptz,
  reject_reason text not null default '',
  updated_at timestamptz not null default now(),
  primary key (driver_id, work_date)
);

-- 이전 버전 Supabase에도 적용되도록 has_permission(), is_admin(),
-- is_active_driver() 함수에 의존하지 않는 전용 권한 확인 함수를 만듭니다.
create or replace function public.departure_can_manage()
returns boolean
language sql
stable
security definer
set search_path = public
as $$
  select exists (
    select 1
    from public.profiles
    where id = auth.uid()
      and active = true
      and role in (
        'admin',
        'super_admin',
        'dispatch_manager',
        'field_manager'
      )
  );
$$;

create or replace function public.departure_is_active_driver()
returns boolean
language sql
stable
security definer
set search_path = public
as $$
  select exists (
    select 1
    from public.profiles
    where id = auth.uid()
      and active = true
      and role = 'driver'
  );
$$;

revoke all on function public.departure_can_manage() from public;
revoke all on function public.departure_is_active_driver() from public;
grant execute on function public.departure_can_manage() to authenticated;
grant execute on function public.departure_is_active_driver() to authenticated;

alter table public.departure_checks enable row level security;

drop policy if exists departure_select on public.departure_checks;
create policy departure_select
on public.departure_checks
for select
to authenticated
using (
  public.departure_can_manage()
  or driver_id = auth.uid()
);

drop policy if exists departure_driver_insert on public.departure_checks;
create policy departure_driver_insert
on public.departure_checks
for insert
to authenticated
with check (
  (
    driver_id = auth.uid()
    and public.departure_is_active_driver()
    and approved = false
  )
  or public.departure_can_manage()
);

drop policy if exists departure_driver_update on public.departure_checks;
create policy departure_driver_update
on public.departure_checks
for update
to authenticated
using (
  (
    driver_id = auth.uid()
    and approved = false
  )
  or public.departure_can_manage()
)
with check (
  (
    driver_id = auth.uid()
    and approved = false
  )
  or public.departure_can_manage()
);

grant select, insert, update
on table public.departure_checks
to authenticated;

-- 감사 기록 함수가 이미 설치되어 있을 때만 출차 승인 변경 이력을 연결합니다.
do $$
begin
  if to_regprocedure('public.audit_change()') is not null then
    drop trigger if exists audit_departure_checks
      on public.departure_checks;

    create trigger audit_departure_checks
      after insert or update or delete
      on public.departure_checks
      for each row
      execute function public.audit_change();
  end if;
end
$$;

-- Supabase API가 새 테이블을 즉시 인식하도록 스키마 캐시를 갱신합니다.
notify pgrst, 'reload schema';

commit;

-- 아래 결과에 public.departure_checks가 나오면 정상적으로 완료된 것입니다.
select to_regclass('public.departure_checks') as created_table;
