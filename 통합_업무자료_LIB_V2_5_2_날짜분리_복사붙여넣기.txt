const MAX_DELIVERIES = 300;
const MAX_ITEMS_PER_DELIVERY = 100;
const MAX_BOXES_PER_DELIVERY = 100;
const dateKey = (value) => String(value || "").slice(0, 10);

function cleanText(value, max = 500) {
  return String(value ?? "").trim().slice(0, max);
}

function cleanNumber(value, min = 0, max = Number.MAX_SAFE_INTEGER) {
  const number = Number(value);
  if (!Number.isFinite(number)) return min;
  return Math.min(max, Math.max(min, number));
}

function cleanBoolean(value) {
  return value === true;
}

function cleanWeighing(entry = {}) {
  return {
    ...entry,
    id: Number(entry.id) || Date.now(),
    weight: cleanNumber(entry.weight, 0, 9999),
  };
}

function cleanItem(item = {}) {
  let weighings = Array.isArray(item.weighings)
    ? item.weighings.slice(0, 300).map(cleanWeighing).filter((row) => row.weight > 0)
    : [];
  const measured = Number(
    weighings.reduce((sum, row) => sum + row.weight, 0).toFixed(3),
  );
  const tankNumber = Math.floor(cleanNumber(item.tankNumber, 0, 6));
  const tankLoaded = tankNumber > 0 && cleanBoolean(item.tankLoaded);
  const now = new Date().toISOString();
  weighings = weighings.map((row) =>
    tankLoaded
      ? {
          ...row,
          tankVerifiedAt: row.tankVerifiedAt || now,
          tankVerifiedBy: row.tankVerifiedBy || "최고 관리자 수정",
          verifiedTank: tankNumber,
          loadedAt: row.loadedAt || now,
          loadedBy: row.loadedBy || "최고 관리자 수정",
        }
      : { ...row, loadedAt: null, loadedBy: "" },
  );

  return {
    ...item,
    id: Number(item.id) || Date.now(),
    category: cleanText(item.category, 30) || "기타",
    species: cleanText(item.species || item.item, 80) || "품목명 미입력",
    size: cleanText(item.size, 80),
    originCountry: cleanText(item.originCountry || item.origin, 50),
    requestedQty: cleanNumber(item.requestedQty, 0, 999999),
    unit: cleanText(item.unit, 20) || "개",
    actualWeight:
      weighings.length > 0
        ? measured
        : item.actualWeight === "" || item.actualWeight == null
          ? null
          : cleanNumber(item.actualWeight, 0, 999999),
    weighings,
    weighingConfirmed:
      weighings.length > 0 && cleanBoolean(item.weighingConfirmed),
    priceUnit: cleanText(item.priceUnit, 20) || cleanText(item.unit, 20) || "kg",
    unitPrice: cleanNumber(item.unitPrice, 0, 999999999),
    processingFeePerKg: cleanNumber(item.processingFeePerKg, 0, 999999999),
    processing: cleanText(item.processing, 80),
    tankNumber,
    tankLoaded,
    tankLoadedAt: tankLoaded
      ? item.tankLoadedAt || new Date().toISOString()
      : null,
    rubberBand: cleanBoolean(item.rubberBand),
    rubberBandColor: cleanBoolean(item.rubberBand)
      ? cleanText(item.rubberBandColor, 40)
      : "",
    rubberBandQuantity: cleanBoolean(item.rubberBand)
      ? Math.floor(cleanNumber(item.rubberBandQuantity, 0, 9999))
      : 0,
    includedNote: cleanText(item.includedNote, 300),
    checked: cleanBoolean(item.checked),
  };
}

function cleanBox(box = {}, index = 0) {
  return {
    ...box,
    id: Number(box.id) || Date.now() + index,
    boxNumber: index + 1,
    contents: cleanText(box.contents, 300),
    checked: cleanBoolean(box.checked),
  };
}

function cleanDelivery(delivery = {}, workDate) {
  const items = Array.isArray(delivery.items)
    ? delivery.items
        .slice(0, MAX_ITEMS_PER_DELIVERY)
        .map(cleanItem)
    : [];
  const boxes = Array.isArray(delivery.boxes)
    ? delivery.boxes
        .slice(0, MAX_BOXES_PER_DELIVERY)
        .map(cleanBox)
    : [];
  const loadedBoxCount = Math.floor(
    cleanNumber(delivery.loadedBoxCount, 0, 9999),
  );
  const boxCountConfirmed = cleanBoolean(delivery.boxCountConfirmed);

  return {
    ...delivery,
    id: Number(delivery.id) || Date.now(),
    customerId: Number(delivery.customerId) || 0,
    fallbackStoreName: cleanText(delivery.fallbackStoreName, 100),
    deliveryDate: workDate,
    routeOrder: Math.max(
      1,
      Math.floor(cleanNumber(delivery.routeOrder, 1, 9999)),
    ),
    status: delivery.status === "cancelled" ? "cancelled" : "ready",
    completedAt: null,
    previousBalance: cleanNumber(delivery.previousBalance, 0, 999999999999),
    notes: cleanText(delivery.notes, 1000),
    coPackNote: cleanText(delivery.coPackNote, 1000),
    isUrgentAdditional: cleanBoolean(delivery.isUrgentAdditional),
    urgentAcknowledged: cleanBoolean(delivery.urgentAcknowledged),
    loadedBoxCount,
    boxCountConfirmed,
    boxCountConfirmedAt: boxCountConfirmed
      ? delivery.boxCountConfirmedAt || new Date().toISOString()
      : null,
    items,
    boxes,
  };
}

function cleanCompletion(value, deliveryCount) {
  if (!value || value.enabled !== true) return null;
  const at = new Date(value.at);
  return {
    at: Number.isNaN(at.getTime()) ? new Date().toISOString() : at.toISOString(),
    result: value.result === "issue" ? "issue" : "normal",
    note: cleanText(value.note, 1000),
    by: cleanText(value.by, 80),
    vehicle: cleanText(value.vehicle, 20),
    deliveryCount,
    urgentPending: Math.floor(cleanNumber(value.urgentPending, 0, 9999)),
    issueCount: Math.floor(cleanNumber(value.issueCount, 0, 9999)),
    lastSyncAt: new Date().toISOString(),
    editedByAdmin: true,
  };
}

function summary(payload, workDate) {
  const deliveries = (payload?.deliveries || []).filter(
    (row) =>
      dateKey(row.deliveryDate) === dateKey(workDate) &&
      row.status !== "cancelled",
  );
  return {
    deliveries: deliveries.length,
    items: deliveries.reduce((sum, row) => sum + (row.items || []).length, 0),
    boxes: deliveries.reduce((sum, row) => sum + (row.boxes || []).length, 0),
    completion: Boolean(payload?.workClosures?.[workDate]),
  };
}

export async function updateDriverWorkday(auth, request, response) {
  try {
    const driverId = cleanText(request.body?.driverId, 80);
    const workDate = cleanText(request.body?.workDate, 10);
    const reason = cleanText(request.body?.reason, 500);
    const inputDeliveries = request.body?.deliveries;
    const customerNames = request.body?.customerNames;

    if (!driverId || !/^\d{4}-\d{2}-\d{2}$/.test(workDate)) {
      return response
        .status(400)
        .json({ error: "배송원과 업무 날짜를 확인해 주세요." });
    }
    if (!reason) {
      return response
        .status(400)
        .json({ error: "관리자 수정 사유를 반드시 입력해 주세요." });
    }
    if (!Array.isArray(inputDeliveries) || inputDeliveries.length > MAX_DELIVERIES) {
      return response
        .status(400)
        .json({ error: "당일 배송 자료의 형식이나 수량을 확인해 주세요." });
    }

    const { data: driver, error: driverError } = await auth.client
      .from("profiles")
      .select("id,role,full_name,vehicle_number")
      .eq("id", driverId)
      .eq("role", "driver")
      .maybeSingle();
    if (driverError || !driver) {
      return response
        .status(404)
        .json({ error: "배송원 계정을 찾지 못했습니다." });
    }

    const { data: snapshot, error: snapshotError } = await auth.client
      .from("driver_state_snapshots")
      .select("driver_id,work_date,payload,updated_at")
      .eq("driver_id", driverId)
      .eq("work_date", workDate)
      .maybeSingle();
    if (snapshotError) throw snapshotError;
    if (!snapshot?.payload) {
      return response.status(404).json({
        error:
          "해당 날짜의 배송원 중앙 저장 자료가 없습니다. 배송원 웹앱에서 한 번 저장한 뒤 다시 시도해 주세요.",
      });
    }

    const beforePayload = snapshot.payload;
    const payload = JSON.parse(JSON.stringify(beforePayload));
    const otherDeliveries = Array.isArray(payload.deliveries)
      ? payload.deliveries.filter(
          (row) => dateKey(row.deliveryDate) !== dateKey(workDate),
        )
      : [];
    const cleanedDeliveries = inputDeliveries.map((row) =>
      cleanDelivery(row, workDate),
    );
    payload.deliveries = [...otherDeliveries, ...cleanedDeliveries];

    if (customerNames && typeof customerNames === "object") {
      payload.customers = Array.isArray(payload.customers)
        ? payload.customers.map((customer) => {
            const nextName = customerNames[String(customer.id)];
            return nextName == null
              ? customer
              : { ...customer, name: cleanText(nextName, 100) };
          })
        : [];
    }

    const activeDeliveries = cleanedDeliveries.filter(
      (row) => row.status !== "cancelled",
    );
    payload.workClosures =
      payload.workClosures && typeof payload.workClosures === "object"
        ? payload.workClosures
        : {};
    const completion = cleanCompletion(
      request.body?.completion,
      activeDeliveries.length,
    );
    if (completion) {
      completion.by = completion.by || driver.full_name;
      completion.vehicle = completion.vehicle || String(driver.vehicle_number || "");
      payload.workClosures[workDate] = completion;
    } else {
      delete payload.workClosures[workDate];
    }

    const now = new Date().toISOString();
    const { error: updateError } = await auth.client
      .from("driver_state_snapshots")
      .update({ payload, updated_at: now })
      .eq("driver_id", driverId)
      .eq("work_date", workDate);
    if (updateError) throw updateError;

    const tankItems = activeDeliveries.flatMap((row) =>
      (row.items || []).filter((item) => Number(item.tankNumber) > 0),
    );
    const boxDeliveries = activeDeliveries.filter(
      (row) => (row.boxes || []).length > 0,
    );
    const boxesChecked = boxDeliveries.reduce(
      (sum, row) =>
        sum +
        (row.boxCountConfirmed && row.loadedBoxCount === (row.boxes || []).length
          ? 1
          : 0),
      0,
    );
    const statusRow = {
      driver_id: driverId,
      work_date: workDate,
      vehicle_number: driver.vehicle_number,
      work_state: completion ? "completed" : "working",
      delivery_total: activeDeliveries.length,
      delivery_completed: completion ? activeDeliveries.length : 0,
      tank_total: tankItems.length,
      tank_loaded: tankItems.filter((item) => item.tankLoaded).length,
      box_total: boxDeliveries.length,
      box_checked: boxesChecked,
    };
    const { error: statusError } = await auth.client
      .from("driver_daily_status")
      .upsert(statusRow, { onConflict: "driver_id,work_date" });
    if (statusError) throw statusError;

    await auth.client.from("activity_logs").insert({
      actor_id: auth.user.id,
      driver_id: driverId,
      action: "driver_workday_updated",
      entity_type: "driver_state_snapshots",
      entity_id: driverId,
      detail: {
        workDate,
        reason,
        before: summary(beforePayload, workDate),
        after: summary(payload, workDate),
      },
    });

    return response.status(200).json({
      ok: true,
      updatedAt: now,
      summary: summary(payload, workDate),
    });
  } catch (error) {
    return response.status(500).json({
      error:
        error instanceof Error ? error.message : "서버 오류가 발생했습니다.",
    });
  }
}

export async function resetDepartureApproval(auth, request, response) {
  try {
    const driverId = cleanText(request.body?.driverId, 80);
    const workDate = cleanText(request.body?.workDate, 10);
    const reason = cleanText(request.body?.reason, 500);
    if (!driverId || !/^\d{4}-\d{2}-\d{2}$/.test(workDate) || !reason) {
      return response
        .status(400)
        .json({ error: "배송원, 날짜와 승인 취소 사유를 입력해 주세요." });
    }

    const { data: current, error: currentError } = await auth.client
      .from("departure_checks")
      .select("*")
      .eq("driver_id", driverId)
      .eq("work_date", workDate)
      .maybeSingle();
    if (currentError) throw currentError;
    if (!current) {
      return response
        .status(404)
        .json({ error: "출차 점검 기록을 찾지 못했습니다." });
    }

    const { error } = await auth.client
      .from("departure_checks")
      .update({
        approved: false,
        approved_by: null,
        approved_at: null,
        reject_reason: reason,
        updated_at: new Date().toISOString(),
      })
      .eq("driver_id", driverId)
      .eq("work_date", workDate);
    if (error) throw error;

    await auth.client.from("activity_logs").insert({
      actor_id: auth.user.id,
      driver_id: driverId,
      action: "departure_approval_reset",
      entity_type: "departure_checks",
      entity_id: driverId,
      detail: { workDate, reason, previousApprovedAt: current.approved_at },
    });

    return response.status(200).json({ ok: true });
  } catch (error) {
    return response.status(500).json({
      error:
        error instanceof Error ? error.message : "서버 오류가 발생했습니다.",
    });
  }
}
