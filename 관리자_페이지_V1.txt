            const stateMatch=filter==="all" || (filter==="online"&&isOnline) || (filter==="working"&&!["not_started","completed"].includes(s.work_state)) || (filter==="completed"&&s.work_state==="completed") || (filter==="disabled"&&!d.active);
            return match&&stateMatch;
          });
        }
