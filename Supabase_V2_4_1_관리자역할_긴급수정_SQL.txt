-- 장백수산 관리자 V2.4.1
-- profiles_role_check 오류를 없애는 긴급 수정 SQL
--
-- 실행 위치:
-- Supabase → SQL Editor → New query
--
-- 사용 방법:
-- 이 파일의 내용을 처음부터 끝까지 전부 복사 → 붙여넣기 → Run
-- 여러 번 실행해도 안전합니다.

begin;

alter table public.profiles
  add column if not exists menu_permissions text[];

-- 과거 버전의 역할 제한과 이전 V2 역할 제한을 모두 찾아 제거합니다.
do $$
declare constraint_row record;
begin
  for constraint_row in
    select conname
    from pg_constraint
    where conrelid = 'public.profiles'::regclass
      and contype = 'c'
      and (
        conname in ('profiles_role_check', 'profiles_role_v2_check')
        or pg_get_constraintdef(oid) ilike '%role%'
      )
  loop
    execute format(
      'alter table public.profiles drop constraint if exists %I',
      constraint_row.conname
    );
  end loop;
end $$;

-- 현재 프로그램이 사용하는 역할 7종을 허용합니다.
alter table public.profiles
  add constraint profiles_role_v2_check
  check (
    role in (
      'admin',
      'super_admin',
      'dispatch_manager',
      'accounting_manager',
      'field_manager',
      'viewer',
      'driver'
    )
  ) not valid;

alter table public.profiles
  validate constraint profiles_role_v2_check;

comment on column public.profiles.menu_permissions is
  '관리자 화면에 표시할 메뉴 ID 배열. NULL이면 역할별 추천 기본값 사용';

commit;

-- 확인 결과
-- constraint_name 칸에 profiles_role_v2_check가 나오면 성공입니다.
select
  conname as constraint_name,
  pg_get_constraintdef(oid) as allowed_roles
from pg_constraint
where conrelid = 'public.profiles'::regclass
  and contype = 'c'
  and pg_get_constraintdef(oid) ilike '%role%'
order by conname;

-- 기존 관리자 계정도 함께 확인합니다.
select
  id,
  employee_code,
  full_name,
  role,
  menu_permissions,
  active
from public.profiles
where role <> 'driver'
order by created_at;
